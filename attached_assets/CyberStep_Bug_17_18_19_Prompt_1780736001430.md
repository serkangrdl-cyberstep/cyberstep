// Shadow IT bileşen işleme kısmını bul.
// Her bileşen için versiyon bilgisi var mı kontrol et.

interface ShadowITComponent {
  name: string;       // "jQuery", "WordPress"
  version: string | null; // "3.7.1" veya null
  isOutdated: boolean | null; // null = bilinmiyor
  riskLevel: "high" | "medium" | "low" | "unknown";
}

// Versiyon tespiti:
// HTTP response body veya HTML meta'dan çek

async function detectJQueryVersion(
  html: string
): Promise<string | null> {
  // jquery/x.x.x CDN URL'inden
  const cdnMatch = html.match(
    /jquery[.-](\d+\.\d+(?:\.\d+)?)(\.min)?\.js/i
  );
  if (cdnMatch) return cdnMatch[1];

  // $.fn.jquery değişkeninden (inline script'te)
  const inlineMatch = html.match(
    /jquery.*?version['":\s]+['"]([\d.]+)['"]/i
  );
  if (inlineMatch) return inlineMatch[1];

  return null; // Tespit edilemedi
}

async function detectWordPressVersion(
  domain: string,
  html: string
): Promise<string | null> {
  // Meta generator tag
  const metaMatch = html.match(
    /name=["']generator["'] content=["']WordPress\s*([\d.]+)/i
  );
  if (metaMatch) return metaMatch[1];

  // readme.html (varsa)
  try {
    const { default: axios } = await import("axios");
    const res = await axios.get(
      `https://${domain}/readme.html`,
      { timeout: 5000, validateStatus: () => true }
    );
    if (res.status === 200) {
      const vMatch = res.data.match(/Version\s*([\d.]+)/i);
      if (vMatch) return vMatch[1];
    }
  } catch { /* erişilemedi */ }

  return null;
}

// Eski versiyon kontrolü:
// (güncel sürümler: jQuery 3.7+, WordPress 6.5+)
const LATEST_VERSIONS: Record<string, string> = {
  jquery:    "3.7.0",
  wordpress: "6.5.0",
};

function isOutdated(
  name: string,
  version: string | null
): boolean | null {
  if (!version) return null; // bilinmiyor
  const latest = LATEST_VERSIONS[name.toLowerCase()];
  if (!latest) return null;
  return compareVersions(version, latest) < 0;
}

// MITRE prompt'una gönderilecek shadow IT context:

  }
  return `${c.name} — versiyon tespit edilemedi`;
}).join("\n")}

## BUG 19 — WAF/CDN TESPİTİ EKSİK (DOLAYLI İŞARETLER)

**Sorun:** netsys.com.tr'de WAF var ama
sistem tespit edemiyor. Sadece bilinen
header'lara bakıyor (x-fortigate, cf-ray vb.)
WAF bunları gizliyorsa tespit edilemiyor.

**Düzeltme:**

```typescript
// wafDetector.ts içindeki tespit fonksiyonunu bul
// Mevcut header kontrollerine ek olarak
// dolaylı işaretleri ekle:

// 1. IP KONUM ANALİZİ
// Türk şirket + yurt dışı IP = CDN/proxy işareti
async function detectCDNByIPLocation(
  ipCountry: string | null,
  ipOrg: string | null,
  domain: string
): Promise<string | null> {

  // Şirket Türkiye'deyken IP yurt dışındaysa
  // büyük ihtimalle CDN kullanıyor
  const domainIsTR = domain.endsWith(".tr");
  const ipIsAbroad = ipCountry && ipCountry !== "TR";

  if (domainIsAbroad && ipIsAbroad) {
    // Hangi ülke?
    const CDN_COUNTRIES: Record<string, string> = {
      "US": "possible_us_cdn",   // Cloudflare, Fastly
      "FR": "possible_ovh_cdn",  // OVH, Gcore
      "DE": "possible_eu_cdn",   // Hetzner CDN
      "NL": "possible_eu_cdn",   // Leaseweb
      "GB": "possible_uk_cdn",
    };
    return CDN_COUNTRIES[ipCountry] || "possible_cdn";
  }
  return null;
}

// 2. DOLAYLI HEADER İŞARETLERİ
const CDN_INDIRECT_HEADERS: Record<string, string> = {
  "cf-ray":          "cloudflare",
  "cf-cache-status": "cloudflare",
  "x-cache":         "cdn_generic",
  "x-served-by":     "fastly",
  "x-cache-hits":    "fastly",
  "via":             "cdn_generic",
  "x-amz-cf-id":     "aws_cloudfront",
  "x-azure-ref":     "azure_cdn",
  "x-sucuri-id":     "sucuri",
  "x-fw":            "fortinet",
};

function detectWAFFromIndirectHeaders(
  headers: Record<string, string>
): string | null {
  for (const [header, provider] of
       Object.entries(CDN_INDIRECT_HEADERS)) {
    if (headers[header.toLowerCase()]) {
      return provider;
    }
  }
  return null;
}

// 3. TÜM TESPİT MANTIĞI (sıralı)
export async function detectWAFEnhanced(
  domain: string,
  httpHeaders: Record<string, string>,
  shodanData: any
): Promise<{
  provider: string | null;
  confidence: "high" | "medium" | "low" | "none";
  note: string | null;
}> {

  // Yöntem 1: Bilinen WAF header'ları (yüksek güven)
  const directProvider = detectWAFFromDirectHeaders(
    httpHeaders
  );
  if (directProvider) {
    return { provider: directProvider,
             confidence: "high", note: null };
  }

  // Yöntem 2: Dolaylı header işaretleri (orta güven)
  const indirectProvider = detectWAFFromIndirectHeaders(
    httpHeaders
  );
  if (indirectProvider) {
    return { provider: indirectProvider,
             confidence: "medium", note: null };
  }

  // Yöntem 3: IP konum analizi (düşük güven)
  const locationHint = await detectCDNByIPLocation(
    shodanData?.country_code,
    shodanData?.org,
    domain
  );
  if (locationHint) {
    return {
      provider: null,
      confidence: "low",
      note:
        "WAF/CDN header'ları tespit edilemedi. " +
        "Sunucu IP'si yurt dışında — " +
        "CDN kullanılıyor olabilir. " +
        "Varsa güvenlik yapılandırmasını " +
        "manuel doğrulayın.",
    };
  }

  return { provider: null, confidence: "none", note: null };
}

// Raporda "note" varsa göster:
// "ℹ️ WAF/CDN header'ları tespit edilemedi.
//    Sunucu IP'si Fransa'da — CDN kullanılıyor
//    olabilir."
```

// portRiskClassifier.ts içinde:

function adjustPortRiskForWAF(
  portRisk: PortRisk,
  wafResult: { provider: string | null;
               confidence: string; }
): PortRisk {

  // Yüksek güvenle tespit edildiyse risk düşür
  if (wafResult.confidence === "high" &&
      wafResult.provider) {
    return downgradePortRisk(portRisk, "cdn_protected");
  }

  // Düşük güven → riski değiştirme, sadece not ekle
  if (wafResult.confidence === "low") {
    portRisk.note =
      "CDN kullanılıyor olabilir. " +
      "Port riskleri CDN arkasında geçerli olmayabilir.";
  }

  return portRisk;
}
