## SERVİS TANIMI

### Ne Değil

```
❌ Sanal CISO değil
❌ Danışmanlık değil
❌ Telefona çıkan biri değil
❌ "CISO'nuzun yerini alır" değil
```

### Ne

```
✅ CISO'nuzun rutin işlerini otomatikleştirir
✅ Board raporunu hazırlar, siz sunarsınız
✅ Politika taslakları üretir, siz onaylarsınız
✅ Uyum durumunu takip eder, siz kararı verirsiniz
✅ Tehditleri özetler, siz aksiyon alırsınız

"CISO'nuz strateji düşünsün,
 rutin raporlamayı biz yapalım."
```

### Hedef Müşteri

```
CISO'su olan ama zamanı kısıtlı şirketler
  → CISO her ay board raporu hazırlamaktan
    bıkmış, bunu otomatikleştirmek istiyor

CISO'su olmayan ama olması gereken şirketler
  → 7545 kapsamında Siber Güvenlik Sorumlusu
    atamak zorunda, ama tam zamanlı bütçesi yok
    → Bu paket "minimum uyum" sağlar

IT direktörü CISO rolü de üstlenen şirketler
  → Raporlama yükünü hafifletir
```

// src/ciso/complianceCalculator.ts

export async function calculateComplianceScore(
  customerId: number
): Promise<ComplianceResult> {

  const sub = await getCISOSubscription(customerId);
  const customer = await getCustomer(customerId);
  const latestScan = await getLatestScan(customer.domain);

  // ─── 7545 KANUNU KONTROL LİSTESİ ─────────────────────
  const checklist7545 = [
    {
      id: 'security_officer',
      label: 'Siber Güvenlik Sorumlusu atandı',
      weight: 25,
      // 50+ çalışanda zorunlu
      passed: sub.hasDedicatedCiso || sub.cisoName !== null,
      note: sub.hasDedicatedCiso
        ? null
        : '7545 Madde 6: 50+ çalışanlı şirketlerde zorunlu.',
    },
    {
      id: 'annual_audit',
      label: 'Yıllık güvenlik denetimi yapıldı',
      weight: 20,
      passed: latestScan !== null,
      // CyberStep taraması = denetim belgesi
      note: latestScan
        ? `Son denetim: ${formatDate(latestScan.createdAt)}`
        : 'CyberStep değerlendirmesi denetim belgesi olarak kullanılabilir.',
    },
    {
      id: 'incident_response',
      label: 'Olay müdahale planı mevcut',
      weight: 20,
      passed: sub.hasIncidentResponsePlan,
      note: sub.hasIncidentResponsePlan
        ? null
        : 'Politika kütüphanesinden olay müdahale prosedürünü indirin.',
    },
    {
      id: 'ssl_valid',
      label: 'SSL sertifikası geçerli',
      weight: 15,
      passed: latestScan?.sslValid === true,
      note: null,
    },
    {
      id: 'dmarc_active',
      label: 'E-posta güvenliği (DMARC) aktif',
      weight: 10,
      passed: latestScan?.dmarcPolicy !== null &&
               latestScan?.dmarcPolicy !== 'none',
      note: null,
    },
    {
      id: 'no_critical_cve',
      label: 'Kritik CVE açığı yok',
      weight: 10,
      passed: (latestScan?.criticalCveCount || 0) === 0,
      note: null,
    },
  ];

  const score7545 = Math.round(
    checklist7545.reduce((sum, item) =>
      sum + (item.passed ? item.weight : 0), 0
    )
  );

  // ─── KVKK KONTROL LİSTESİ ────────────────────────────
  const checklistKVKK = [
    {
      id: 'verbis',
      label: 'VERBİS kaydı yapıldı',
      weight: 25,
      passed: sub.kvkkVerbisRegistered,
      note: sub.kvkkVerbisRegistered
        ? null
        : 'verbis.kvkk.gov.tr adresinden kayıt yapılmalı.',
    },
    {
      id: 'data_inventory',
      label: 'Kişisel veri envanteri mevcut',
      weight: 20,
      passed: sub.hasDataInventory,
      note: null,
    },
    {
      id: 'no_breach',
      label: 'Bilinen veri sızıntısı yok',
      weight: 25,
      passed: (latestScan?.hibpBreachCount || 0) === 0,
      note: latestScan?.hibpBreachCount > 0
        ? `${latestScan.hibpBreachCount} sızıntı tespit edildi.`
        : null,
    },
    {
      id: 'breach_notification',
      label: '72 saat bildirim kapasitesi var',
      weight: 15,
      // CyberStep CVE + IOC sistemi bu kapasiteyi sağlıyor
      passed: true,
      note: 'CyberStep anlık tehdit bildirimi ile karşılanıyor.',
    },
    {
      id: 'technical_measures',
      label: 'Teknik tedbirler yeterli (SSL, DMARC)',
      weight: 15,
      passed: (latestScan?.overallScore || 0) >= 60,
      note: null,
    },
  ];

  const scoreKVKK = Math.round(
    checklistKVKK.reduce((sum, item) =>
      sum + (item.passed ? item.weight : 0), 0
    )
  );

  // DB'ye kaydet
  const month = new Date().toISOString().slice(0, 7);
  await db.insert(complianceScores).values({
    customerId,
    scoreMonth: month,
    score7545,
    scoreKvkk: scoreKVKK,
    checklist7545,
    checklistKvkk: checklistKVKK,
  }).onConflictDoUpdate({
    target: [complianceScores.customerId,
             complianceScores.scoreMonth],
    set: { score7545, scoreKvkk: scoreKVKK,
           checklist7545, checklistKvkk: checklistKVKK }
  });

  return { score7545, scoreKVKK,
           checklist7545, checklistKVKK };
}

// src/ciso/boardReportGenerator.ts

export async function generateBoardReport(
  customerId: number
): Promise<void> {

  const customer = await getCustomer(customerId);
  const sub = await getCISOSubscription(customerId);
  const latestScan = await getLatestScan(customer.domain);
  const prevMonthScan = await getLastMonthScan(customer.domain);
  const compliance = await calculateComplianceScore(customerId);
  const sectorAvg = await getSectorAverageScore(customer.sector);

  // Finansal risk tahmini
  // Ortalama Türkiye veri ihlali maliyeti (2024 IBM raporu baz alındı)
  const BASE_BREACH_COST_TRY = 8_500_000;
  const riskMultiplier = Math.max(0,
    (100 - (latestScan?.overallScore || 50)) / 100
  );
  const financialRiskLow = Math.round(
    BASE_BREACH_COST_TRY * riskMultiplier * 0.15
  );
  const financialRiskHigh = Math.round(
    BASE_BREACH_COST_TRY * riskMultiplier * 0.5
  );

  // Top 3 öneri
  const recommendations = buildRecommendations(
    latestScan, compliance
  );

  // Claude ile yönetici özeti (Sonnet — kaliteli olsun)
  const executiveSummary = await callClaude(`
Şirket: ${customer.companyName}
Sektör: ${customer.sector}
Dönem: ${new Date().toLocaleDateString('tr-TR', { month: 'long', year: 'numeric' })}

Güvenlik skoru: ${latestScan?.overallScore}/100
Geçen ay: ${prevMonthScan?.overallScore || 'veri yok'}
Sektör ortalaması: ${sectorAvg}/100
Kritik bulgu: ${latestScan?.criticalCount || 0}
7545 uyumu: %${compliance.score7545}
KVKK uyumu: %${compliance.scoreKVKK}
Tahmini finansal risk: ${financialRiskLow.toLocaleString('tr-TR')} — ${financialRiskHigh.toLocaleString('tr-TR')} TL

Yönetim kuruluna sunulacak 3 paragraflık özet yaz.
CEO okuyacak — teknik kelime yok.
Birinci paragraf: genel durum
İkinci paragraf: bu ay öne çıkan risk
Üçüncü paragraf: önerilen aksiyon
Maksimum 150 kelime.
  `, {
    model: 'claude-sonnet-4-6',
    // Board raporu kaliteli olsun — Sonnet kullan
    maxTokens: 300,
    system: 'Türkiye şirketi için Türkçe siber güvenlik yönetim raporu yaz. Sade, net, patron dili.',
  });

  // PDF oluştur
  const pdfPath = await generateBoardReportPDF({
    customer,
    riskScore: latestScan?.overallScore || 0,
    riskScorePrev: prevMonthScan?.overallScore || 0,
    sectorAvgScore: sectorAvg,
    criticalFindings: latestScan?.criticalCount || 0,
    closedFindings: await getClosedFindingsCount(customerId),
    compliance7545: compliance.score7545,
    complianceKvkk: compliance.scoreKVKK,
    financialRiskLow,
    financialRiskHigh,
    recommendations,
    executiveSummary,
    reportMonth: new Date().toLocaleDateString('tr-TR',
      { month: 'long', year: 'numeric' }),
  });

  // DB'ye kaydet
  const month = new Date().toISOString().slice(0, 7);
  await db.insert(boardReports).values({
    customerId,
    reportMonth: month,
    riskScore: latestScan?.overallScore || 0,
    complianceKvkk: compliance.scoreKVKK,
    compliance7545: compliance.score7545,
    financialRiskLow,
    financialRiskHigh,
    topRecommendations: recommendations,
    executiveSummary,
    pdfPath,
    status: 'draft',
  }).onConflictDoUpdate({
    target: [boardReports.customerId, boardReports.reportMonth],
    set: { executiveSummary, pdfPath,
           status: 'draft', generatedAt: new Date() }
  });

  // Email gönder
  await sendBoardReportEmail(
    sub.boardReportEmail || customer.email,
    customer.companyName,
    pdfPath,
    month
  );
}

export async function generatePolicyLibrary(
  customerId: number
): Promise<void> {

  const sub = await getCISOSubscription(customerId);
  const customer = await getCustomer(customerId);

  for (const [policyType, template] of Object.entries(POLICY_TEMPLATES)) {
    // Zaten oluşturulmuş mu?
    const existing = await db.select()
      .from(securityPolicies)
      .where(
        and(
          eq(securityPolicies.customerId, customerId),
          eq(securityPolicies.policyType, policyType)
        )
      ).limit(1);

    if (existing[0]) continue;

    const prompt = template.prompt
      .replace(/{COMPANY}/g, customer.companyName)
      .replace(/{SECTOR}/g, customer.sector || 'teknoloji')
      .replace(/{EMPLOYEE_COUNT}/g,
        sub.employeeCount?.toString() || 'belirtilmemiş');

    const content = await callClaude(prompt, {
      model: 'claude-haiku-4-5',
      maxTokens: 800,
      system: 'Türkiye şirketi için Türkçe güvenlik politikası yaz. Profesyonel, uygulanabilir, 7545 ve KVKK uyumlu.',
    });

    // DOCX oluştur
    const docxPath = await generatePolicyDocx(
      template.title, content, customer.companyName
    );

    await db.insert(securityPolicies).values({
      customerId,
      policyType,
      title: template.title,
      content,
      docxPath,
      status: 'draft',
    });

    await sleep(500);
  }

  await db.update(cisoAssistantSubscriptions).set({
    policiesGeneratedAt: new Date(),
    policiesCount: Object.keys(POLICY_TEMPLATES).length,
  }).where(eq(cisoAssistantSubscriptions.customerId, customerId));
}
