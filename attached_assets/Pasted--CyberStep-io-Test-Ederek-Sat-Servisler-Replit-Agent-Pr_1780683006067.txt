```typescript
// GET /api/portal/integrations/microsoft365/guide
router.get("/microsoft365/guide", requireCustomer,
  async (req, res) => {
  res.json(MICROSOFT_365_GUIDE);
});

// POST /api/portal/integrations/microsoft365/connect
router.post("/microsoft365/connect", requireCustomer,
  async (req, res) => {

  const { tenantId, clientId, clientSecret } = req.body;
  const customerId = getCustomerId(req)!;

  // Bağlantı testi
  const testResult = await testMicrosoft365Connection({
    tenantId, clientId, clientSecret
  });

  if (!testResult.success) {
    res.status(400).json({
      error: "Bağlantı başarısız",
      detail: testResult.error,
      troubleshooting:
        MICROSOFT_365_GUIDE.troubleshooting,
    });
    return;
  }

  // Şifreli kaydet
  await saveIntegrationConfig(customerId, "microsoft365", {
    tenantId,
    clientId,
    clientSecret: encrypt(clientSecret),
  });

  res.json({
    success: true,
    message: "Microsoft 365 bağlantısı kuruldu. " +
             "İlk güvenlik verileri 15 dakika içinde gelecek.",
  });
});
```

```typescript
// Değerlendirme tamamlandığında rapor üret
export async function generateAssessmentReport(
  customerId: number,
  answers: Record<string, string | string[]>
): Promise<void> {

  const score = calculateAssessmentScore(answers);

  // Claude Sonnet ile detaylı rapor
  const report = await callClaude(`
Siber güvenlik olgunluk değerlendirmesi raporu yaz.

Şirket bilgileri:
  Sektör: ${answers.sector}
  Çalışan sayısı: ${answers.employee_count}
  Genel skor: ${score.total}/100

Bölüm skorları:
  Altyapı: ${score.infrastructure}/100
  Kimlik & Erişim: ${score.identity}/100
  Veri Güvenliği: ${score.data}/100
  Ağ Güvenliği: ${score.network}/100
  KVKK Uyum: ${score.kvkk}/100
  Olay Müdahale: ${score.incident}/100

Kritik yanıtlar:
${JSON.stringify(score.criticalAnswers, null, 2)}

Şunları üret:
1. Yönetici özeti (CEO dili, 200 kelime)
2. En kritik 5 risk (madde madde)
3. 30-60-90 gün aksiyon planı
4. 7545 Kanunu uyum boşlukları
5. KVKK teknik tedbir eksiklikleri

Türkçe, profesyonel, somut öneriler.
  `, {
    model: "claude-sonnet-4-6",
    maxTokens: 2000,
  });

  // PDF üret ve müşteriye gönder
  const pdf = await generateAssessmentPDF(score, report);

  await sendEmail({
    to: await getCustomerEmail(customerId),
    subject:
      "Güvenlik Değerlendirme Raporunuz Hazır — CyberStep",
    html: buildAssessmentEmail(score, pdf.url),
    attachments: [{ filename: pdf.filename, path: pdf.path }],
    from: "degerlendirme@cyberstep.io",
  });

  // Notlar:
  // 1. İlk 2-3 satışta sen raporu gözden geçir
  // 2. Claude'un ürettiği aksiyonlar
  //    müşterinin tech stack'ine göre
  //    ne kadar isabetli, gör
  // 3. Skor hesabını kalibre et
}
```

```typescript
// Rapor üretimi
export async function generatePhishingAnalysis(
  customerId: number,
  params: PhishingAnalysisParams
): Promise<void> {

  // 1. OSINT — domain hakkında kamuya açık bilgi
  const osintData = await collectDomainOSINT(params.domain);

  // 2. Claude ile saldırı senaryoları üret
  const scenarios = await callClaude(`
${params.domain} domainli ${params.sector}
sektöründeki bir şirketi hedef alan
phishing saldırı senaryoları üret.

OSINT verileri:
  Çalışan sayısı: ${params.employeeCount}
  Kamuya açık LinkedIn profilleri: ${osintData.linkedinCount}
  Email formatı: ${osintData.emailFormat || "bilinmiyor"}
  CEO/yönetici adları: ${osintData.executives?.join(", ") || "bulunamadı"}

5 ayrı senaryo üret:
  1. CEO fraud / yönetici taklidi
  2. Fatura/ödeme sahteciliği
  3. IT desteği taklidi
  4. Tedarikçi/partner taklidi
  5. ${params.knownRisks?.[0] || "Genel phishing"}

Her senaryo için:
  - Saldırı yöntemi
  - Örnek email konusu (gerçekçi)
  - Hedef çalışan tipi
  - Başarı ihtimali (Düşük/Orta/Yüksek)
  - Tespit ipuçları
  - Çalışana verilecek eğitim notu

Türkçe, gerçekçi, eğitim amaçlı.
  `, {
    model: "claude-haiku-4-5",
    maxTokens: 1500,
  });

  // 3. Farkındalık raporu üret + PDF gönder
  const pdf = await generatePhishingReportPDF(
    params, scenarios, osintData
  );

  await sendEmail({
    to: await getCustomerEmail(customerId),
    subject: "AI Phishing Risk Analiz Raporunuz — CyberStep",
    html: buildPhishingReportEmail(pdf.url),
    attachments: [{ filename: pdf.filename, path: pdf.path }],
    from: "analiz@cyberstep.io",
  });
}
```

```typescript
// GET /api/portal/onboarding/status
router.get("/status", requireCustomer, async (req, res) => {
  const customerId = getCustomerId(req)!;
  const subs = await getActiveSubscriptions(customerId);

  const status = await Promise.all(
    subs.map(async sub => {
      const guide = getServiceGuide(sub.serviceName);
      const completed =
        await getOnboardingProgress(customerId, sub.serviceName);
      return {
        serviceName: sub.serviceName,
        serviceLabel: sub.serviceLabel,
        hasGuide: !!guide,
        guideUrl: guide
          ? `${BASE_URL}/portal/integrations/${sub.serviceName}/guide`
          : null,
        completionPct: completed.pct,
        nextStep: completed.nextStep,
      };
    })
  );

  res.json({ services: status });
});
```
