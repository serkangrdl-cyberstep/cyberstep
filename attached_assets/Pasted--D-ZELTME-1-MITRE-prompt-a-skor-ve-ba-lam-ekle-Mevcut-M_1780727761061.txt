// DÜZELTME 3 — Skor-MITRE tutarlılık kontrolü
// Rapor üretilmeden önce kontrol ekle:

function validateReportConsistency(report: ScanReport): void {
  // Yüksek skor + yüksek tehdit = çelişki
  if (report.overallScore >= 80 &&
      report.mitreTheatLevel === "Yüksek") {
    logger.warn({
      domain: report.domain,
      score: report.overallScore,
      mitre: report.mitreTheatLevel,
    }, "MITRE tehdit seviyesi skora göre çok yüksek — düzeltiliyor");

    report.mitreTheatLevel = "Düşük";
  }

  if (report.overallScore >= 60 &&
      report.mitreTheatLevel === "Kritik") {
    report.mitreTheatLevel = "Orta";
  }
}
